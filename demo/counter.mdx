# Counter

<Init initialState={{ count: 0 }}>

<Observe>
  {({ setState }) => (
    <button onClick={() => setState(s => ({ count: s.count + 1 }))}>
      Click me
    </button>
  )}
</Observe>

<Observe>
  { ({...state}) => (<span>{state.count} times</span>) }
</Observe>

</Init>
